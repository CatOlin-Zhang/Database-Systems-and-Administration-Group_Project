-- Create database
DROP DATABASE IF EXISTS ecommerce_platform;
CREATE DATABASE ecommerce_platform;
USE ecommerce_platform;

-- ============================================================
-- 1. Create Base Tables
-- ============================================================

-- 1.1 Vendor Table
CREATE TABLE Vendor (
    vendor_id INT AUTO_INCREMENT PRIMARY KEY,
    business_name VARCHAR(100) NOT NULL UNIQUE,
    avg_rating DECIMAL(3,2) DEFAULT 0.00,
    geo_location VARCHAR(200) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 1.2 Customer Table
CREATE TABLE Customer (
    customer_id INT AUTO_INCREMENT PRIMARY KEY,
    contact_num VARCHAR(20) NOT NULL UNIQUE,
    shipping_address VARCHAR(255) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 1.3 Tag Table
CREATE TABLE Tag (
    tag_id INT AUTO_INCREMENT PRIMARY KEY,
    tag_name VARCHAR(50) NOT NULL UNIQUE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 1.4 Product Table
CREATE TABLE Product (
    product_id INT AUTO_INCREMENT PRIMARY KEY,
    product_name VARCHAR(100) NOT NULL,
    price DECIMAL(10,2) NOT NULL CHECK (price > 0),
    stock_quantity INT NOT NULL DEFAULT 0 CHECK (stock_quantity >= 0),
    vendor_id INT NOT NULL,
    CONSTRAINT fk_product_vendor FOREIGN KEY (vendor_id) 
        REFERENCES Vendor(vendor_id) ON DELETE CASCADE,
    CONSTRAINT uk_product_vendor UNIQUE (product_name, vendor_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 1.5 Orders Table
CREATE TABLE Orders (
    order_id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT NOT NULL,
    order_date DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    total_price DECIMAL(10,2) NOT NULL DEFAULT 0.00 CHECK (total_price >= 0),
    order_status ENUM('PENDING_PAY', 'PENDING_SHIP', 'SHIPPED', 'COMPLETED', 'CANCELLED') 
        NOT NULL DEFAULT 'PENDING_PAY',
    CONSTRAINT fk_orders_customer FOREIGN KEY (customer_id) 
        REFERENCES Customer(customer_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 1.6 ProductTag Table (Associative Entity)
CREATE TABLE ProductTag (
    protag_id INT AUTO_INCREMENT PRIMARY KEY,
    product_id INT NOT NULL,
    tag_id INT NOT NULL,
    CONSTRAINT fk_producttag_product FOREIGN KEY (product_id) 
        REFERENCES Product(product_id) ON DELETE CASCADE,
    CONSTRAINT fk_producttag_tag FOREIGN KEY (tag_id) 
        REFERENCES Tag(tag_id) ON DELETE CASCADE,
    CONSTRAINT uk_product_tag UNIQUE (product_id, tag_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 1.7 OrderItem Table (Associative Entity)
CREATE TABLE OrderItem (
    item_id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    product_id INT NOT NULL,
    buy_quantity INT NOT NULL CHECK (buy_quantity >= 1),
    unit_price DECIMAL(10,2) NOT NULL CHECK (unit_price > 0),
    CONSTRAINT fk_orderitem_order FOREIGN KEY (order_id) 
        REFERENCES Orders(order_id) ON DELETE CASCADE,
    CONSTRAINT fk_orderitem_product FOREIGN KEY (product_id) 
        REFERENCES Product(product_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 1.8 Transaction Table
CREATE TABLE Transaction (
    transaction_id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    vendor_id INT NOT NULL,
    pay_amount DECIMAL(10,2) NOT NULL CHECK (pay_amount > 0),
    transaction_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_transaction_order FOREIGN KEY (order_id) 
        REFERENCES Orders(order_id) ON DELETE CASCADE,
    CONSTRAINT fk_transaction_vendor FOREIGN KEY (vendor_id) 
        REFERENCES Vendor(vendor_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 1.9 Rating Table
CREATE TABLE Rating (
    rating_id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT NOT NULL,
    vendor_id INT NOT NULL,
    score INT NOT NULL CHECK (score >= 1 AND score <= 5),
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_rating_customer FOREIGN KEY (customer_id) 
        REFERENCES Customer(customer_id) ON DELETE CASCADE,
    CONSTRAINT fk_rating_vendor FOREIGN KEY (vendor_id) 
        REFERENCES Vendor(vendor_id) ON DELETE CASCADE,
    CONSTRAINT uk_customer_vendor UNIQUE (customer_id, vendor_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ============================================================

CREATE INDEX idx_orderitem_order ON OrderItem(order_id);
CREATE INDEX idx_orderitem_product ON OrderItem(product_id);

CREATE INDEX idx_transaction_order ON Transaction(order_id);
CREATE INDEX idx_transaction_vendor ON Transaction(vendor_id);

-- Rating table indexes
CREATE INDEX idx_rating_vendor ON Rating(vendor_id);
CREATE INDEX idx_rating_customer ON Rating(customer_id);

-- ProductTag table indexes
CREATE INDEX idx_producttag_product ON ProductTag(product_id);
CREATE INDEX idx_producttag_tag ON ProductTag(tag_id);

-- ============================================================
-- 3. Create Views
-- ============================================================

-- 3.1 Product View (with tags and vendor info)
CREATE VIEW ProductFullView AS
SELECT 
    p.product_id,
    p.product_name,
    p.price,
    p.stock_quantity,
    v.vendor_id,
    v.business_name AS vendor_name,
    v.avg_rating AS vendor_rating,
    GROUP_CONCAT(DISTINCT t.tag_name ORDER BY t.tag_name SEPARATOR ', ') AS tags
FROM Product p
JOIN Vendor v ON p.vendor_id = v.vendor_id
LEFT JOIN ProductTag pt ON p.product_id = pt.product_id
LEFT JOIN Tag t ON pt.tag_id = t.tag_id
GROUP BY p.product_id;

-- 3.2 Order Details View
CREATE VIEW OrderDetailView AS
SELECT 
    o.order_id,
    o.order_date,
    o.order_status,
    o.total_price,
    c.customer_id,
    c.contact_num,
    c.shipping_address,
    oi.item_id,
    oi.product_id,
    p.product_name,
    oi.buy_quantity,
    oi.unit_price,
    (oi.buy_quantity * oi.unit_price) AS item_total
FROM Orders o
JOIN Customer c ON o.customer_id = c.customer_id
JOIN OrderItem oi ON o.order_id = oi.order_id
JOIN Product p ON oi.product_id = p.product_id;

-- 3.3 Vendor Sales Statistics View
CREATE VIEW VendorSalesView AS
SELECT 
    v.vendor_id,
    v.business_name,
    COUNT(DISTINCT t.transaction_id) AS total_transactions,
    SUM(t.pay_amount) AS total_sales,
    AVG(r.score) AS avg_rating
FROM Vendor v
LEFT JOIN Transaction t ON v.vendor_id = t.vendor_id
LEFT JOIN Rating r ON v.vendor_id = r.vendor_id
GROUP BY v.vendor_id;

-- 3.4 Customer Spending Statistics View
CREATE VIEW CustomerStatsView AS
SELECT 
    c.customer_id,
    c.contact_num,
    COUNT(DISTINCT o.order_id) AS total_orders,
    SUM(o.total_price) AS total_spent,
    MAX(o.order_date) AS last_order_date
FROM Customer c
LEFT JOIN Orders o ON c.customer_id = o.customer_id
WHERE o.order_status != 'CANCELLED'
GROUP BY c.customer_id;

-- 4. Stored Procedures and Triggers
-- ============================================================

DELIMITER //

-- 4.1 Stored Procedure: Update vendor average rating
CREATE PROCEDURE UpdateVendorAvgRating(IN p_vendor_id INT)
BEGIN
    UPDATE Vendor v
    SET v.avg_rating = (
        SELECT COALESCE(AVG(score), 0)
        FROM Rating r 
        WHERE r.vendor_id = p_vendor_id
    )
    WHERE v.vendor_id = p_vendor_id;
END //

-- 4.2 Stored Procedure: Create order with transaction
CREATE PROCEDURE CreateOrder(
    IN p_customer_id INT,
    IN p_items_json JSON
)
BEGIN
    DECLARE v_order_id INT;
    DECLARE v_total DECIMAL(10,2) DEFAULT 0;
    DECLARE v_idx INT DEFAULT 0;
    DECLARE v_count INT;
    DECLARE v_product_id INT;
    DECLARE v_quantity INT;
    DECLARE v_unit_price DECIMAL(10,2);
    DECLARE v_vendor_id INT;
    DECLARE done INT DEFAULT FALSE;
    
    DECLARE cur CURSOR FOR 
        SELECT product_id, quantity 
        FROM JSON_TABLE(p_items_json, '$[*]' COLUMNS(
            product_id INT PATH '$.product_id',
            quantity INT PATH '$.quantity'
        )) AS jt;
    
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;
    
    START TRANSACTION;
    
    -- Create order
    INSERT INTO Orders (customer_id, order_date, order_status, total_price)
    VALUES (p_customer_id, NOW(), 'PENDING_PAY', 0);
    SET v_order_id = LAST_INSERT_ID();
    
    -- Insert order items and update stock
    OPEN cur;
    read_loop: LOOP
        FETCH cur INTO v_product_id, v_quantity;
        IF done THEN
            LEAVE read_loop;
        END IF;
        
        -- Get product price and vendor
        SELECT price, vendor_id INTO v_unit_price, v_vendor_id
        FROM Product WHERE product_id = v_product_id;
        
        -- Check stock
        IF (SELECT stock_quantity FROM Product WHERE product_id = v_product_id) < v_quantity THEN
            SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Insufficient stock';
        END IF;
        
        -- Insert order item
        INSERT INTO OrderItem (order_id, product_id, buy_quantity, unit_price)
        VALUES (v_order_id, v_product_id, v_quantity, v_unit_price);
        
        -- Deduct stock
        UPDATE Product 
        SET stock_quantity = stock_quantity - v_quantity 
        WHERE product_id = v_product_id;
        
        -- Accumulate total
        SET v_total = v_total + (v_unit_price * v_quantity);
    END LOOP;
    CLOSE cur;
    
    -- Update order total
    UPDATE Orders SET total_price = v_total WHERE order_id = v_order_id;
    
    -- Create transaction records per vendor
    INSERT INTO Transaction (order_id, vendor_id, pay_amount, transaction_time)
    SELECT 
        v_order_id,
        p.vendor_id,
        SUM(oi.buy_quantity * oi.unit_price),
        NOW()
    FROM OrderItem oi
    JOIN Product p ON oi.product_id = p.product_id
    WHERE oi.order_id = v_order_id
    GROUP BY p.vendor_id;
    
    COMMIT;
END //

-- 4.3 Stored Procedure: Cancel order
CREATE PROCEDURE CancelOrder(IN p_order_id INT)
BEGIN
    DECLARE v_status VARCHAR(20);
    
    -- Check order status
    SELECT order_status INTO v_status FROM Orders WHERE order_id = p_order_id;
    
    IF v_status IN ('PENDING_PAY', 'PENDING_SHIP') THEN
        START TRANSACTION;
        
        -- Restore stock
        UPDATE Product p
        JOIN OrderItem oi ON p.product_id = oi.product_id
        SET p.stock_quantity = p.stock_quantity + oi.buy_quantity
        WHERE oi.order_id = p_order_id;
        
        -- Cancel order
        UPDATE Orders SET order_status = 'CANCELLED' WHERE order_id = p_order_id;
        
        -- Delete related transactions
        DELETE FROM Transaction WHERE order_id = p_order_id;
        
        COMMIT;
    ELSE
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Order cannot be cancelled';
    END IF;
END //

-- 4.4 Trigger: Update vendor rating after insert
CREATE TRIGGER after_rating_insert
AFTER INSERT ON Rating
FOR EACH ROW
BEGIN
    CALL UpdateVendorAvgRating(NEW.vendor_id);
END //

-- 4.5 Trigger: Update vendor rating after update
CREATE TRIGGER after_rating_update
AFTER UPDATE ON Rating
FOR EACH ROW
BEGIN
    IF OLD.score != NEW.score THEN
        CALL UpdateVendorAvgRating(NEW.vendor_id);
    END IF;
END //

DELIMITER ;
